enum CourseLevel{BEGINNER, INTERMEDIATE, ADVANCED
};
string getCourseLevelName(CourseLevel level){
    switch(level){
        case BEGINNER: return "Beginner";
        case INTERMEDIATE: return "Intermediate";
        case ADVANCED: return "Advanced";
        default: return "Unknown";
    }
}

double getGradePoint(double marks){
	if(marks >= 84.5){
	    return 4.00;
    }	
    if(marks >= 79.5){
    	return 3.70;
	}
	if(marks >= 74.5){
		return 3.40;
	}
	if(marks >= 69.5){
		return 3.00;
	}
	if(marks >= 64.5){
		return 2.50;
	}
	if(marks >= 59.5){
		return 2.00;
	}
	if(marks >= 54.5){
		return 1.50;
	}
	if(marks >= 49.5){
		return 1.00;
	}
	    return 0.00;
    
}

class Course{            //COURSE CLASS
	public:
		string code;
		string title;
		string schedule;
		int credits;
		CourseLevel level;
		vector<string>prerequisites;
		
		Course(){           //DEFAULT CONSTRUCTOR
		}
		Course(string c, string t, int cr, string s, CourseLevel l) : code(c), title(t), credits(cr), schedule(s), level(l){      //PARAMETERIZED CONSTRUCTOR
		}                   
		                    //(DEFAULT + PARAMETERIZED CONSTRUCTOR) = CONSTRUCTOR OVERLOADING
		                    
	void addPrerequisite(string p){
		prerequisites.push_back(p);
	}	                    
		void display()const{
			cout<<"Code: "<<code<<endl;
			cout<<"Title: "<<title<<endl;
			cout<<"Credits: "<<credits<<endl;
			cout<<"Schedule: "<<schedule<<endl;
			cout<<"Level: "<<getCourseLevelName(level)<<endl;
			
		if(!prerequisites.empty()){
			cout<<"Prerequisites: ";
			
			for(string p: prerequisites){
			cout<<p<<" ";
			cout<<endl;
		    }
		}	
			
	}
};
                               //ABSTRACT CLASS(USER)
class user{                         
	protected:
		string name;
		int id;
		
	public:
		user(){       //DEFAULT CONSTRUCTOT
		}
	user(string n, int i) : name(n), id(i){         //PARAMETERIZED
	}	
	virtual void displayInfo() = 0;     //PURE VIRTUAL FUNCTION
	virtual void accessPortal() = 0;    //PURE VIRTUAL FUNCTION
	
	int getId()const{return id;
	}
	string getName()const{return name;
	}

};
                                    //STUDENT CLASS
class Student : public user{                 
    private:
    string contact;
	vector<string>registeredCourses;
	map<string, double>courseGrades;
	
	public:
		Student(){
		}
		Student(string n, int i, string c) : user(n, i), contact(c){
		}
		void registerCourse(string courseCode){
			registeredCourses.push_back(courseCode);
		}
		
		void addGrade(string courseCode, double grade){
			courseGrades[courseCode] = grade;
		}
		
		double calculateGPA(map<string, Course>& courseMap){
			double totalPoints = 0;
			int totalCredits = 0;
			
		for(string code : registeredCourses){
		
			if(courseGrades.count(code) && courseMap.count(code)){
				double marks = courseGrades[code];
				int credits = courseMap[code].credits;
					
					totalPoints += getGradePoint(marks)*credits;
					totalCredits += credits;
				}
			}
	
			return(totalCredits == 0) ? 0.0 : totalPoints/totalCredits;
	}
		
		void displayInfo(){
			cout<<"Student Name: "<<name<<endl;
			cout<<"ID: "<<id<<endl;
			cout<<"Contact: "<<contact<<endl;
			
		}
		void accessPortal(){
			cout<<"Registered Courses: "<<endl;
			for(string c : registeredCourses){
			cout<<c<<" ";
			cout<<endl;
		    }
		}
		
		vector<string>getRegisteredCourses(){
			return registeredCourses;
		}


};

class Faculty : public user{              //FACULTY CLASS
	private:
		string email;
		vector<string>assignedCourses;
		map<int, map<string, double> >studentGrades;
		
		public:
			Faculty(){
			}
			Faculty(string n, int i, string e) : user(n, i), email(e){
			}
			
		void assignCourses(string courseCode){
				assignedCourses.push_back(courseCode);
			}
			
		void gradeStudent(int studentId, string courseCode, double grade){
				studentGrades[studentId][courseCode] = grade;
			}
			
		void displayInfo(){
				cout<<"Faculty Name: "<<name<<endl;
				cout<<"ID: "<<id<<endl;
				cout<<"Email: "<<email<<endl;
			}
		void accessPortal(){
			cout<<"Assigned Courses: "<<endl;
			for(string c : assignedCourses){
			cout<<c<<" ";
			cout<<endl;
		    }
	}
	    vector<string>getAssignedCourses(){
			return assignedCourses;
		}
	
};

	                               //ENTER GRADES
	void enterGrades(){
		int fid;
		
		cout<<"Enter Faculty ID: "<<endl;

		if(!allFaculties.count(fid)){
			cout<<"Faculty not found."<<endl;
			return;
		}
		
		Faculty& f = allFaculties[fid];
		vector<string>courses = f.getAssignedCourses();
		
		for(auto& pair : allStudents){
			Student& s = pair.second;
			cout<<"Enter marks for students: "<<s.getName()<<"(ID: "<<s.getId()<<"):"<<endl;
			
			for(string course : courses){
				double marks;
				
				cout<<"---"<<course<<": ";
				cin>>marks;
				
				s.addGrade(course, marks);
				f.gradeStudent(s.getId(), course, marks);
			}
		}
	}
